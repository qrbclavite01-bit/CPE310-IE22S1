import pandas as pd

df = pd.read_csv('/content/netflix_titles.csv')

df.shape

df.head()

df.tail()

"""**2. Data Cleaning. Which columns contain missing values? How might you handle missing entries? Drop them or fill with a placeholder?**"""

df.isnull().sum()

df['director'] = df['director'].fillna('Unknown')
df['cast'] = df['cast'].fillna('Unknown')
df['country'] = df['country'].fillna('Unknown')

df.dropna(subset = ['date_added', 'rating', 'duration'], inplace = True)

df['date_added']=pd.to_datetime(df['date_added'].str.strip())

df['rating'] = df['rating'].str.strip()

df['duration'] = df['duration'].str.extract(r'(\d+)').astype(float)

print(df.isnull().sum())

"""**3. Data Types & Conversion. Check each column’s data type. Are there any that you need to convert (like date to datetime type)?**"""

df.dtypes

df = df.copy()
df['date_added'] = df['date_added'].apply(pd.to_datetime)
df.head()



"""**4. Univariate Analysis. How many Movies vs. TV Shows are there? What are the top five most common ratings? Which release year appears most frequently?**"""

type_counts = df['type'].value_counts()
print(type_counts)

df['rating'].value_counts().head(5)

df['release_year'].value_counts().head(1)

"""**5. Duration & Seasons. Parse duration into numeric minutes (for movies) and number of seasons (for shows). What is the average movie length? The average number of seasons per TV show?**"""

movies = df[df['type'] == 'Movie']
shows  = df[df['type'] == 'TV Show']

avg_movie_mins   = movies['duration'].mean()
avg_show_seasons = shows['duration'].mean()

print(f'Average movie length:          {avg_movie_mins:.1f} minutes')
print(f'Average seasons per TV show:   {avg_show_seasons:.1f} seasons')

"""**6. Genre Analysis. Which genre has the highest average release year? What might that indicate?**"""

import matplotlib.pyplot as plt

genre_df = df.assign(genre=df['listed_in'].str.split(', ')).explode('genre')
genre_df['genre'] = genre_df['genre'].str.strip()

genre_avg_year = genre_df.groupby('genre')['release_year'].mean().sort_values(ascending=False)
print(genre_avg_year.head(10))

"""**7. Temporal Trends. Are there spikes or drops? Compare release_year vs. year_added: how long after release does Netflix typically acquire content?**

"""

df['year_added'] = df['date_added'].dt.year
df['gap_years']  = df['year_added'] - df['release_year']

print(f'Average gap (release → Netflix): {df[df["gap_years"] >= 0]["gap_years"].mean():.1f} years')
print(f'Median  gap:                     {df[df["gap_years"] >= 0]["gap_years"].median():.1f} years')

added_per_year = df.groupby('year_added').size()

added_per_year.plot(kind='bar', figsize=(12, 5), color='coral', edgecolor='white')
plt.title('Number of Titles Added to Netflix per Year')
plt.xlabel('Year Added')
plt.ylabel('Count')
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()

print()

"""8. Rating vs. Type. Create a cross-tab or pivot table of type vs. rating. Which rating dominates movies compared to TV shows?"""

crosstab = pd.crosstab(df['rating'], df['type'])
print(crosstab)

import seaborn as sns

plt.figure(figsize=(10, 6))
sns.heatmap(crosstab, annot=True, fmt='d', cmap='Blues', linewidths=0.5)
plt.title('Rating vs. Content Type')
plt.tight_layout()
plt.show()

"""**9. Filtering & Querying. Extract all titles with rating "R" added after 2020. How many?**"""

r_after_2020 = df[(df['rating'] == 'R') & (df['year_added'] > 2020)]

print(f'Count: {len(r_after_2020)}')
r_after_2020[['title', 'type', 'rating', 'year_added']].head(10)

"""**10. Aggregations & GroupBy. For each country, compute the average release year of its titles. Group by year_added and compute the proportion of Movies vs. TV Shows added each year.**"""

df['primary_country'] = df['country'].str.split(',').str[0].str.strip()

country_avg = (df[df['primary_country'] != 'Unknown']
               .groupby('primary_country')['release_year']
               .mean()
               .sort_values(ascending=False))

country_avg.head(10)

type_year = df.groupby(['year_added', 'type']).size().unstack(fill_value=0)
type_year_pct = type_year.div(type_year.sum(axis=1), axis=0) * 100

print(type_year_pct.round(1))

type_year_pct.plot(kind='bar', stacked=True, figsize=(12, 5),
                   color=['steelblue', 'tomato'], edgecolor='white')
plt.title('Proportion of Movies vs. TV Shows Added Each Year')
plt.xlabel('Year Added')
plt.ylabel('Percentage (%)')
plt.xticks(rotation=45)
plt.legend(title='Type')
plt.tight_layout()
plt.show()

"""**11. Applying Functions. Write a function that, given a director's name, returns their titles sorted by release_year. Create a reusable function to plot the top N categories for any categorical column.**"""

def get_director_titles(director_name):
    mask = df['director'].str.contains(director_name, case=False, na=False)
    return (df[mask][['title', 'type', 'release_year', 'rating', 'director']]
              .sort_values('release_year')
              .reset_index(drop=True))

get_director_titles('Steven Spielberg')

def plot_top_n_categories(column, n=10):
    counts = df[column].value_counts().head(n)
    counts.plot(kind='barh', figsize=(10, 5), color='mediumpurple', edgecolor='white')
    plt.title(f'Top {n} — {column}')
    plt.xlabel('Count')
    plt.gca().invert_yaxis()
    plt.tight_layout()
    plt.show()

plot_top_n_categories('rating')

plot_top_n_categories('primary_country', n=15)
